## 1. 嵌套 if 语句

当需要多重条件判断时，可以使用嵌套 if：

```c
#include <stdio.h>

int main() {
    int score = 85;
    char attendance = 'Y';
    
    if (attendance == 'Y') {
        if (score >= 90) {
            printf("优秀且全勤！\n");
        } else if (score >= 60) {
            printf("及格且全勤\n");
        } else {
            printf("缺考或不及格\n");
        }
    } else {
        printf("缺勤\n");
    }
    
    return 0;
}
```

## 2. 逻辑运算符

使用 && (与)、|| (或)、! (非) 组合条件：

```c
#include <stdio.h>

int main() {
    int age = 20;
    int score = 85;
    
    // 使用逻辑运算符
    if (age >= 18 && score >= 60) {
        printf("成年且及格\n");
    }
    
    if (age < 18 || score < 60) {
        printf("未成年或不及格\n");
    }
    
    return 0;
}
```

## 3. break 和 continue

在循环中控制流程：

```c
#include <stdio.h>

int main() {
    int i;
    
    // break 示例：找到第一个能被3和5整除的数就停止
    for (i = 1; i <= 100; i++) {
        if (i % 3 == 0 && i % 5 == 0) {
            printf("找到第一个能被3和5整除的数: %d\n", i);
            break;  // 跳出循环
        }
    }
    
    // continue 示例：跳过偶数，只打印奇数
    printf("1-10中的奇数: ");
    for (i = 1; i <= 10; i++) {
        if (i % 2 == 0) {
            continue;  // 跳过本次循环的剩余部分
        }
        printf("%d ", i);
    }
    printf("\n");
    
    return 0;
}
```

## 4. 综合练习

### 成绩评级系统：
```c
#include <stdio.h>

int main() {
    int score;
    
    printf("请输入成绩 (0-100): ");
    scanf("%d", &score);
    
    if (score < 0 || score > 100) {
        printf("输入错误！成绩应在0-100之间\n");
    } else if (score >= 90) {
        printf("等级: A\n");
    } else if (score >= 80) {
        printf("等级: B\n");
    } else if (score >= 70) {
        printf("等级: C\n");
    } else if (score >= 60) {
        printf("等级: D\n");
    } else {
        printf("等级: F (不及格)\n");
    }
    
    return 0;
}
```
